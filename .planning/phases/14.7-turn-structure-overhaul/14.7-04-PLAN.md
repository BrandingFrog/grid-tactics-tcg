```python
def _deploy_minion(
    state: GameState,
    action: Action,
    card_def: CardDefinition,
    card_numeric_id: int,
    active_side: PlayerSide,
    library: CardLibrary,
) -> GameState:
    """Push a summon_declaration originator onto the stack; the minion lands later.

    Validation (deployment zone, occupancy) still runs here. If validation passes,
    we capture the deploy position + card into an originator and enter REACT with
    AFTER_SUMMON_DECLARATION context. The minion is NOT placed on the board yet —
    it lands in resolve_summon_declaration_originator after the react window closes
    (unless the declaration is negated).

    Mana was already spent by the caller (resolve_action's PLAY_CARD branch) BEFORE
    _deploy_minion was invoked; that mana does NOT refund on Window A negate.
    """
    deploy_pos = action.position
    if deploy_pos is None:
        raise ValueError("Minion deployment requires a position")
    row, col = deploy_pos

    # Validate deployment zone (unchanged)
    if active_side == PlayerSide.PLAYER_1:
        friendly_rows = PLAYER_1_ROWS
        back_row = BACK_ROW_P1
    else:
        friendly_rows = PLAYER_2_ROWS
        back_row = BACK_ROW_P2

    if card_def.attack_range == 0:
        if row not in friendly_rows:
            raise ValueError(f"Melee minion must deploy to friendly rows {friendly_rows}, got row {row}")
    else:
        if row != back_row:
            raise ValueError(f"Ranged minion must deploy to back row {back_row}, got row {row}")

    if state.board.get(row, col) is not None:
        raise ValueError(f"Cell ({row}, {col}) is occupied")

    # Push summon_declaration originator
    from grid_tactics.react_stack import ReactEntry
    from grid_tactics.enums import ReactContext
    originator = ReactEntry(
        player_idx=state.active_player_idx,
        card_index=-1,  # already removed from hand by caller
        card_numeric_id=card_numeric_id,
        target_pos=tuple(deploy_pos),
        is_originator=True,
        origin_kind="summon_declaration",
        source_minion_id=None,  # minion doesn't exist yet
        effect_payload=None,    # no effects at declaration stage
    )
    new_stack = state.react_stack + (originator,)
    return replace(
        state,
        react_stack=new_stack,
        phase=TurnPhase.REACT,
        react_player_idx=1 - state.active_player_idx,
        react_context=ReactContext.AFTER_SUMMON_DECLARATION,
        react_return_phase=TurnPhase.ACTION,
    )
```

```python
def resolve_summon_declaration_originator(
    state: GameState, entry: ReactEntry, library: CardLibrary,
) -> GameState:
    """Window A resolved without negate: land the minion on the board.

    If the minion has any ON_SUMMON effects, push a summon_effect originator
    onto the stack and re-enter REACT (the caller's resolve_react_stack loop
    will continue processing). If no ON_SUMMON effects, just return state;
    the stack unwinds naturally.

    Called ONLY when this entry was NOT negated (negated_indices skip happens
    in the caller).
    """
    from grid_tactics.enums import EffectType, ReactContext, TriggerType
    from grid_tactics.minion import MinionInstance

    card_def = library.get_by_id(entry.card_numeric_id)
    owner_side = state.players[entry.player_idx].side
    pos = entry.target_pos
    if pos is None:
        # Shouldn't happen — deploy validation ensured pos != None
        return state

    row, col = pos
    # Verify cell still empty — another effect could have occupied it during the react chain.
    # If occupied, the summon fizzles (minion doesn't land). This is a fizzle-like edge case
    # that should be rare enough to tolerate; proper §7 fizzle rule lands in 14.7-06.
    if state.board.get(row, col) is not None:
        return state

    minion = MinionInstance(
        instance_id=state.next_minion_id,
        card_numeric_id=entry.card_numeric_id,
        owner=owner_side,
        position=pos,
        current_health=card_def.health,
        from_deck=True,
    )
    new_board = state.board.place(row, col, minion.instance_id)
    state = replace(
        state,
        board=new_board,
        minions=state.minions + (minion,),
        next_minion_id=state.next_minion_id + 1,
    )

    # Check if minion has any ON_SUMMON effects → open Window B
    on_summon_effects = [
        (i, e) for i, e in enumerate(card_def.effects)
        if e.trigger == TriggerType.ON_SUMMON
    ]
    if not on_summon_effects:
        return state

    # Build effect_payload for Window B (same shape as magic_cast)
    effects_payload = tuple(
        (i, None, int(owner_side))  # target_pos=None: Summon: triggers don't use action.target_pos
        for (i, _) in on_summon_effects
    )
    effect_originator = ReactEntry(
        player_idx=entry.player_idx,
        card_index=-1,
        card_numeric_id=entry.card_numeric_id,
        target_pos=None,
        is_originator=True,
        origin_kind="summon_effect",
        source_minion_id=minion.instance_id,
        effect_payload=effects_payload,
    )

    # Push onto stack + re-enter REACT
    new_stack = state.react_stack + (effect_originator,)
    return replace(
        state,
        react_stack=new_stack,
        phase=TurnPhase.REACT,
        react_player_idx=1 - state.active_player_idx,
        react_context=ReactContext.AFTER_SUMMON_EFFECT,
        react_return_phase=TurnPhase.ACTION,
    )
```

IMPORTANT ARCHITECTURAL NOTE — re-entering REACT while INSIDE resolve_react_stack's LIFO loop:
resolve_react_stack iterates `stack_list = list(reversed(state.react_stack))` once at the top. If we mutate state.react_stack mid-iteration via this helper, the iterator doesn't know about the new entry.

Two options:
(a) After this helper returns, the outer resolve_react_stack SHOULD re-invoke itself on the new state if phase==REACT. That's a recursion-like pattern. Simpler: have resolve_react_stack check at the end — if phase==REACT with a pending new entry, return state early (don't advance to return_phase). The new React window then opens naturally for the opponent.
(b) Process the declaration originator EARLIER — hoist the land-and-push-window-B logic OUT of the LIFO loop so it runs AFTER the loop but BEFORE the phase transition.

Go with option (a): at the bottom of resolve_react_stack, AFTER the LIFO loop but BEFORE the phase dispatch, check:

```python
# If resolution pushed a new originator (summon_effect Window B), the phase is already REACT
# with a fresh window open. Return early — the caller (events.py) will drive the new react window.
if state.phase == TurnPhase.REACT and len(state.react_stack) > 0:
    return state
```

This ensures Window B opens properly after Window A resolves without negation. NOTE: state.react_stack was cleared earlier in the flow via `replace(state, react_stack=(), ...)` in the phase-dispatch block — but that block runs BELOW this early-return check, so we're safe. Re-read the resolve_react_stack flow to verify the ordering before committing.

```python
def resolve_summon_effect_originator(
    state: GameState, entry: ReactEntry, library: CardLibrary,
) -> GameState:
    """Window B resolved without negate: fire the minion's ON_SUMMON effects.

    Same pattern as magic_cast originator but with ON_SUMMON-filtered effects.
    Dispatches TUTOR / REVIVE via the pending-entry shims identically to _cast_magic
    (see 14.7-01 Task 1 Step 3).
    """
    from grid_tactics.enums import EffectType
    from grid_tactics.effect_resolver import resolve_effect, _enter_pending_tutor
    from grid_tactics.action_resolver import _enter_pending_revive

    card_def = library.get_by_id(entry.card_numeric_id)
    caster_owner = state.players[entry.player_idx].side

    # If the source minion died between declaration and effect resolution (e.g. via a react),
    # the effect still fires from the captured position. True fizzle logic lands in 14.7-06.
    source_pos = None
    if entry.source_minion_id is not None:
        src = state.get_minion(entry.source_minion_id)
        if src is not None:
            source_pos = src.position
    if source_pos is None:
        source_pos = (0, 0)  # fallback — effects that need position will fizzle in 14.7-06

    for (effect_idx, _target_pos_raw, _caster_owner_int) in (entry.effect_payload or ()):
        effect = card_def.effects[effect_idx]
        if effect.effect_type == EffectType.TUTOR:
            state = _enter_pending_tutor(
                state, card_def, caster_owner, library,
                amount=max(1, effect.amount or 1),
            )
        elif effect.effect_type == EffectType.REVIVE:
            state = _enter_pending_revive(state, card_def, caster_owner, library)
        else:
            state = resolve_effect(
                state, effect, source_pos, caster_owner, library, target_pos=None,
            )

    return state
```
