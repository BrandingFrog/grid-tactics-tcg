```python
def fire_start_of_turn_triggers(state: GameState, library: CardLibrary) -> GameState:
    """Enqueue all ON_START_OF_TURN triggers into the pending_trigger_queue_{turn,other}.

    After enqueueing, calls drain_pending_trigger_queue to either auto-resolve
    (if each queue has <=1 entry) or open the modal picker.
    """
    from grid_tactics.game_state import PendingTrigger
    active_side = state.players[state.active_player_idx].side
    other_side = PlayerSide.PLAYER_1 if active_side == PlayerSide.PLAYER_2 else PlayerSide.PLAYER_2

    turn_triggers = []
    other_triggers = []

    ordered_minions = sorted(state.minions, key=lambda m: (m.position[0], m.position[1]))
    for m in ordered_minions:
        card_def = library.get_by_id(m.card_numeric_id)
        for eff_idx, effect in enumerate(card_def.effects):
            if effect.trigger != TriggerType.ON_START_OF_TURN:
                continue
            owner_idx = 0 if m.owner == PlayerSide.PLAYER_1 else 1
            pt = PendingTrigger(
                trigger_kind="start_of_turn",
                source_minion_id=m.instance_id,
                source_card_numeric_id=m.card_numeric_id,
                effect_idx=eff_idx,
                owner_idx=owner_idx,
                captured_position=m.position,
                target_pos=None,
            )
            if m.owner == active_side:
                turn_triggers.append(pt)
            else:
                other_triggers.append(pt)

    if not turn_triggers and not other_triggers:
        return state

    state = replace(
        state,
        pending_trigger_queue_turn=tuple(turn_triggers),
        pending_trigger_queue_other=tuple(other_triggers),
    )
    return drain_pending_trigger_queue(state, library)
```

```python
def drain_pending_trigger_queue(state: GameState, library: CardLibrary) -> GameState:
    """Drain the pending-trigger queues in priority order.

    Turn queue drains first. If >=2 remain in current queue, set picker_idx
    so the UI opens the modal. If exactly 1, auto-resolve. If 0, move to
    the other queue. When both queues are empty, clear picker_idx and return.

    Each auto-resolve or picked-resolve opens its own react window per spec §7.5.
    """
    # If picker modal is already open, do not auto-advance - wait for TRIGGER_PICK.
    if state.pending_trigger_picker_idx is not None:
        return state

    # Try turn queue first, then other queue
    turn_q = state.pending_trigger_queue_turn
    other_q = state.pending_trigger_queue_other
    active_idx = state.active_player_idx
    other_idx = 1 - active_idx

    # Pick the queue to work on: turn queue has priority
    if len(turn_q) >= 2:
        return replace(state, pending_trigger_picker_idx=active_idx)
    elif len(turn_q) == 1:
        # Auto-resolve the single entry, then re-drain
        return _resolve_trigger_and_open_react_window(state, turn_q[0], is_turn_queue=True, library=library)
    elif len(other_q) >= 2:
        return replace(state, pending_trigger_picker_idx=other_idx)
    elif len(other_q) == 1:
        return _resolve_trigger_and_open_react_window(state, other_q[0], is_turn_queue=False, library=library)
    else:
        # Both queues empty - drain complete
        return state


def _resolve_trigger_and_open_react_window(
    state: GameState, trigger: PendingTrigger, is_turn_queue: bool, library: CardLibrary,
) -> GameState:
    """Resolve one PendingTrigger + open a react window per §7.5 step 3.

    The react window fires AFTER each resolution, so other players can react to
    each trigger as it resolves. When that window closes, drain_pending_trigger_queue
    is called again (via resolve_react_stack's new drain-recheck hook in Step 3).
    """
    from grid_tactics.effect_resolver import resolve_effect
    from grid_tactics.action_resolver import _cleanup_dead_minions, _check_game_over
    from grid_tactics.enums import ReactContext

    card_def = library.get_by_id(trigger.source_card_numeric_id)
    effect = card_def.effects[trigger.effect_idx]
    caster_owner = PlayerSide.PLAYER_1 if trigger.owner_idx == 0 else PlayerSide.PLAYER_2

    # Resolve effect (14.7-06 will add fizzle check here)
    state = resolve_effect(
        state, effect, trigger.captured_position, caster_owner, library, trigger.target_pos,
    )
    state = _cleanup_dead_minions(state, library)
    state = _check_game_over(state)
    if state.is_game_over:
        return state

    # Remove the resolved trigger from the appropriate queue
    if is_turn_queue:
        new_turn = state.pending_trigger_queue_turn[1:]  # pop front
        state = replace(state, pending_trigger_queue_turn=new_turn)
    else:
        new_other = state.pending_trigger_queue_other[1:]
        state = replace(state, pending_trigger_queue_other=new_other)

    # Open react window for this trigger's resolution (spec §7.5 step 3)
    if trigger.trigger_kind == "start_of_turn":
        rc = ReactContext.AFTER_START_TRIGGER
    elif trigger.trigger_kind == "end_of_turn":
        rc = ReactContext.BEFORE_END_OF_TURN
    elif trigger.trigger_kind == "on_death":
        rc = ReactContext.AFTER_DEATH_EFFECT
    else:
        rc = ReactContext.AFTER_ACTION

    # Open react window. react_return_phase must preserve the PHASE context so drain continues.
    # Keep the CURRENT phase as react_return_phase so close of this window returns to
    # the same "we are in the middle of start-of-turn draining" state.
    return replace(
        state,
        phase=TurnPhase.REACT,
        react_player_idx=1 - state.active_player_idx,
        react_context=rc,
        react_return_phase=state.phase if state.phase != TurnPhase.REACT else state.react_return_phase,
    )
```

```python
def _apply_trigger_pick(state: GameState, pick_idx: int, library: CardLibrary) -> GameState:
    """Player picks a queue index (0..N-1). Resolve that trigger + open react window."""
    from grid_tactics.react_stack import _resolve_trigger_and_open_react_window
    picker = state.pending_trigger_picker_idx
    active = state.active_player_idx
    is_turn_queue = (picker == active)
    q = state.pending_trigger_queue_turn if is_turn_queue else state.pending_trigger_queue_other
    if pick_idx < 0 or pick_idx >= len(q):
        raise ValueError(f"Invalid pick_idx {pick_idx} for queue of length {len(q)}")

    picked = q[pick_idx]
    # Remove the picked entry from the queue
    new_q = q[:pick_idx] + q[pick_idx+1:]
    if is_turn_queue:
        state = replace(state, pending_trigger_queue_turn=new_q)
    else:
        state = replace(state, pending_trigger_queue_other=new_q)
    # Clear picker - _resolve_trigger will open react window; drain resumes after
    state = replace(state, pending_trigger_picker_idx=None)

    return _resolve_trigger_and_open_react_window(state, picked, is_turn_queue=is_turn_queue, library=library)


def _apply_decline_trigger(state: GameState, library: CardLibrary) -> GameState:
    """Player declines - all remaining triggers in that picker's queue fizzle silently."""
    picker = state.pending_trigger_picker_idx
    active = state.active_player_idx
    is_turn_queue = (picker == active)
    if is_turn_queue:
        state = replace(state, pending_trigger_queue_turn=(), pending_trigger_picker_idx=None)
    else:
        state = replace(state, pending_trigger_queue_other=(), pending_trigger_picker_idx=None)
    # Re-drain (will move to other queue if applicable, or exit)
    from grid_tactics.react_stack import drain_pending_trigger_queue
    return drain_pending_trigger_queue(state, library)
```

```python
def enrich_pending_trigger_for_viewer(state_dict: dict, state: GameState, viewer_idx: int) -> dict:
    """If pending_trigger_picker_idx matches viewer, add serialized queue entries.

    The picker sees the FULL card face for each queue entry's source card (via the
    source_card_numeric_id), so game.js can render via renderDeckBuilderCard.

    Opponent sees a passive "Opponent is picking trigger order..." placeholder via
    pending_trigger_picker_idx being set but no queue contents.
    """
    if state.pending_trigger_picker_idx != viewer_idx:
        return state_dict

    is_turn = (viewer_idx == state.active_player_idx)
    q = state.pending_trigger_queue_turn if is_turn else state.pending_trigger_queue_other
    state_dict["pending_trigger_picker_options"] = [
        {
            "queue_idx": i,
            "trigger_kind": t.trigger_kind,
            "source_card_numeric_id": t.source_card_numeric_id,
            "source_minion_id": t.source_minion_id,
            "owner_idx": t.owner_idx,
        }
        for i, t in enumerate(q)
    ]
    return state_dict
```
