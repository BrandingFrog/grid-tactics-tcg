Step 2 — Rewrite `_cleanup_dead_minions` to enqueue into the priority queues instead of resolving inline:

```python
def _cleanup_dead_minions(state: GameState, library: CardLibrary) -> GameState:
    """Detect dead minions, remove them from board, enqueue their on_death effects.

    14.7-05b: Uses turn-player-first priority queue (pending_trigger_queue_turn/_other)
    instead of the pre-14.7 (active_first, instance_id) inline-resolve sort.

    Flow:
    1. Find all minions with current_health <= 0
    2. For each dead minion, for each ON_DEATH effect on its card_def, build a
       PendingTrigger(trigger_kind='on_death', source_minion_id=..., ...)
    3. Split triggers by owner: turn player's deaths -> queue_turn, other -> queue_other
    4. Remove dead minions from state.minions and state.board
    5. Add their card_numeric_id to the owner's grave
    6. Merge new triggers into existing queues (if something was already queued,
       append - chain-reaction pass preserves priority)
    7. Call drain_pending_trigger_queue to advance

    Note: pending_death_target (effect-level SINGLE_TARGET DESTROY modal) fires
    INSIDE _resolve_trigger_and_open_react_window when an ON_DEATH effect is
    resolving. That mechanism is preserved unchanged; it just runs per trigger
    instead of per-minion inline.
    """
    from grid_tactics.game_state import PendingTrigger
    from grid_tactics.react_stack import drain_pending_trigger_queue
    from grid_tactics.enums import TriggerType

    # 1. Find dead minions
    dead = [m for m in state.minions if m.current_health <= 0]
    if not dead:
        return state

    active_side = state.players[state.active_player_idx].side
    new_turn_triggers = []
    new_other_triggers = []

    # 2. Build PendingTrigger entries for each dead minion's on_death effects
    for m in dead:
        card_def = library.get_by_id(m.card_numeric_id)
        owner_idx = 0 if m.owner == PlayerSide.PLAYER_1 else 1
        for eff_idx, effect in enumerate(card_def.effects):
            if effect.trigger != TriggerType.ON_DEATH:
                continue
            pt = PendingTrigger(
                trigger_kind="on_death",
                source_minion_id=m.instance_id,  # will be dead at resolution time - fizzle if needed
                source_card_numeric_id=m.card_numeric_id,
                effect_idx=eff_idx,
                owner_idx=owner_idx,
                captured_position=m.position,
                target_pos=None,  # DESTROY/SINGLE_TARGET effects use pending_death_target modal
            )
            if m.owner == active_side:
                new_turn_triggers.append(pt)
            else:
                new_other_triggers.append(pt)

    # 3. Remove dead minions from board/state + add to grave
    dead_ids = {m.instance_id for m in dead}
    for m in dead:
        # Add to owner's grave (only if from_deck per existing convention)
        if m.from_deck:
            owner_idx = 0 if m.owner == PlayerSide.PLAYER_1 else 1
            new_grave = state.players[owner_idx].grave + (m.card_numeric_id,)
            new_players = list(state.players)
            new_players[owner_idx] = replace(new_players[owner_idx], grave=new_grave)
            state = replace(state, players=tuple(new_players))
        # Remove from board
        state = replace(state, board=state.board.remove(m.position[0], m.position[1]))

    # Filter dead from minions tuple
    state = replace(
        state,
        minions=tuple(m for m in state.minions if m.instance_id not in dead_ids),
    )

    # 4. Merge new triggers into existing queues (append preserves priority)
    if new_turn_triggers or new_other_triggers:
        state = replace(
            state,
            pending_trigger_queue_turn=state.pending_trigger_queue_turn + tuple(new_turn_triggers),
            pending_trigger_queue_other=state.pending_trigger_queue_other + tuple(new_other_triggers),
        )
        # 5. Drain
        state = drain_pending_trigger_queue(state, library)

    return state
```

```python
def _resolve_trigger_and_open_react_window(
    state: GameState, trigger: PendingTrigger, is_turn_queue: bool, library: CardLibrary,
) -> GameState:
    """14.7-05 base + 14.7-05b death-handling.

    For on_death triggers whose effect is DESTROY + SINGLE_TARGET (e.g. Lasercannon),
    the resolution must open the existing pending_death_target modal instead of
    resolving inline. After the user picks the target via DEATH_TARGET_PICK, the
    effect resolves, and the flow continues to opening the react window.
    """
    from grid_tactics.enums import EffectType, TargetType, ReactContext
    from grid_tactics.effect_resolver import resolve_effect
    from grid_tactics.action_resolver import _check_game_over, _enter_pending_death_target

    card_def = library.get_by_id(trigger.source_card_numeric_id)
    effect = card_def.effects[trigger.effect_idx]
    caster_owner = PlayerSide.PLAYER_1 if trigger.owner_idx == 0 else PlayerSide.PLAYER_2

    # 14.7-05b: Death-effect SINGLE_TARGET DESTROY opens pending_death_target modal
    if (trigger.trigger_kind == "on_death"
        and effect.effect_type == EffectType.DESTROY
        and effect.target_type == TargetType.SINGLE_TARGET):
        # Enter pending_death_target; the user picks a target via DEATH_TARGET_PICK.
        # The resume-after-pick path (existing in action_resolver.py) will resolve the
        # effect and then open the react window + re-drain.
        state = _enter_pending_death_target(state, trigger, card_def, effect, library)
        return state

    # Standard resolve (same as 14.7-05)
    state = resolve_effect(
        state, effect, trigger.captured_position, caster_owner, library, trigger.target_pos,
    )
    # Chain-reaction cleanup: if the resolved effect killed another minion, re-enqueue
    state = _cleanup_dead_minions(state, library)
    state = _check_game_over(state)
    if state.is_game_over:
        return state

    # Remove the resolved trigger from the appropriate queue
    if is_turn_queue:
        new_turn = state.pending_trigger_queue_turn[1:]
        state = replace(state, pending_trigger_queue_turn=new_turn)
    else:
        new_other = state.pending_trigger_queue_other[1:]
        state = replace(state, pending_trigger_queue_other=new_other)

    # Open react window
    if trigger.trigger_kind == "start_of_turn":
        rc = ReactContext.AFTER_START_TRIGGER
    elif trigger.trigger_kind == "end_of_turn":
        rc = ReactContext.BEFORE_END_OF_TURN
    elif trigger.trigger_kind == "on_death":
        rc = ReactContext.AFTER_DEATH_EFFECT
    else:
        rc = ReactContext.AFTER_ACTION

    return replace(
        state,
        phase=TurnPhase.REACT,
        react_player_idx=1 - state.active_player_idx,
        react_context=rc,
        react_return_phase=state.phase if state.phase != TurnPhase.REACT else state.react_return_phase,
    )
```
