```javascript
function _showTurnBanner(turnNumber, activePlayerIdx) {
  // Non-blocking: append a DOM element with CSS keyframe animation, auto-remove after animation ends.
  // Does NOT block AnimationQueue.
  var banner = document.createElement('div');
  banner.className = 'turn-banner';
  banner.innerHTML = '<div class="turn-banner-line1">TURN ' + turnNumber + '</div>' +
                     '<div class="turn-banner-line2">PLAYER ' + (activePlayerIdx + 1) + '</div>';
  document.body.appendChild(banner);
  setTimeout(function() {
    if (banner.parentNode) banner.parentNode.removeChild(banner);
  }, 1800);  // match CSS animation-duration (1.2-1.8s range; pick 1.5s for balance)
}
```
