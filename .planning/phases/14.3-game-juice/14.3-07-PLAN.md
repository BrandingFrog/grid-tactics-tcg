   ```js
   function showFloatingPopup(tileEl, text, variant) {
     if (!tileEl) return;
     var el = document.createElement('div');
     el.className = 'floating-popup ' + variant;
     el.textContent = text;
     tileEl.appendChild(el);
     // Auto-remove after animation. Use animationend with a setTimeout fallback.
     var removed = false;
     var cleanup = function() {
       if (removed) return;
       removed = true;
       if (el.parentNode) el.parentNode.removeChild(el);
     };
     el.addEventListener('animationend', cleanup);
     setTimeout(cleanup, 1400);
   }
   ```

   ```js
   if (gameState && gameState.board && frame.board) {
     // Build a map of minion_id -> prev minion (across both players' boards)
     var prevMinions = {};
     collectMinions(gameState).forEach(function(m){ prevMinions[m.id] = m; });
     collectMinions(frame).forEach(function(m){
       var p = prevMinions[m.id];
       if (!p) return; // newly summoned this frame, no popup
       var prevHp = p.current_health;
       var nextHp = m.current_health;
       var tileEl = getTileElForMinion(m); // existing helper or compute from m.position

       // Heal: HP went UP
       if (nextHp > prevHp) {
         showFloatingPopup(tileEl, '💚 +' + (nextHp - prevHp), 'heal');
       }

       // Burn tick: HP went DOWN AND prev had burning stacks AND active player just flipped
       // We detect "active player flipped" by comparing gameState.active_player vs frame.active_player
       var turnFlipped = gameState.active_player !== frame.active_player;
       if (turnFlipped && p.burning_stacks > 0 && nextHp < prevHp) {
         var dmg = prevHp - nextHp;
         showFloatingPopup(tileEl, '🔥 -' + dmg, 'burn-tick');
       }
     });
   }
   ```
