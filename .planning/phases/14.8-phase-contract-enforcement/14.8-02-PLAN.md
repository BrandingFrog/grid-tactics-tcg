**Add to phase_contracts.py:**

A `ViolationCapture` context manager that intercepts `_log_violation` calls in shadow mode and stores them for inspection:

```python
class ViolationCapture:
    """Context manager that captures phase-contract violations during shadow mode.

    Usage:
        with ViolationCapture() as cap:
            resolve_action(state, some_action, library)
        if cap.violations:
            pytest.fail(format_violations(cap.violations))
    """
    def __init__(self):
        self.violations = []  # list[OutOfPhaseError]

    def __enter__(self):
        global _capture_target
        _capture_target = self
        return self

    def __exit__(self, *args):
        global _capture_target
        _capture_target = None

# Internal: _log_violation appends to _capture_target.violations if set
_capture_target = None

def _log_violation(err):
    logging.getLogger("grid_tactics.phase_contracts").warning(
        "PhaseContractViolation source=%s phase=%s allowed=%s pending_required=%s",
        err.contract_source, err.phase, err.allowed_phases, err.pending_required,
    )
    if _capture_target is not None:
        _capture_target.violations.append(err)
```

Also add a helper `format_violations(violations) -> str` that produces a human-readable bullet list grouping by contract_source and listing distinct phase / pending_required combinations seen.

```python
class TestTriggerInvariants:
    def test_every_card_trigger_only_fires_in_allowed_phases(self, library):
        from grid_tactics.phase_contracts import (
            ViolationCapture, PHASE_CONTRACTS, format_violations,
        )
        for card in library.cards.values():
            for effect in card.effects:
                source = f"trigger:{effect.trigger.name.lower()}"
                allowed = PHASE_CONTRACTS[source]
                for phase in TurnPhase:
                    state = _build_minimal_state_in_phase(library, phase, with_card=card)
                    with ViolationCapture() as cap:
                        try:
                            _fire_trigger_for_card(state, card, effect, library)
                        except Exception:
                            pass  # We're testing contract invariants, not engine correctness
                    if phase in allowed:
                        assert cap.violations == [], (
                            f"Card {card.card_id} trigger {effect.trigger.name} should fire cleanly in {phase.name}, but logged: {format_violations(cap.violations)}"
                        )
                    else:
                        # In strict mode, this should raise; in shadow it should log exactly one violation
                        assert any(v.contract_source == source for v in cap.violations), (
                            f"Card {card.card_id} trigger {effect.trigger.name} fired in DISALLOWED phase {phase.name} but did NOT log a violation. Either the contract table is wrong or the assertion site is missing."
                        )
```

Helper `_build_minimal_state_in_phase(library, phase, with_card)`: constructs a GameState with active_player_idx=0, mana=10 each, fresh decks containing `with_card`, the card placed appropriately for the trigger (in hand for ON_PLAY, on board for ON_DEATH/ON_ATTACK/etc.), and `state.phase = phase`. For START_OF_TURN / END_OF_TURN, also set `state.react_context = None`. For REACT, set react_player_idx and a sensible react_context. Keep this builder small — it's a TEST helper, not production code; if it's too big, split into per-phase factories.

Helper `_fire_trigger_for_card(state, card, effect, library)`: dispatches based on trigger type:
- ON_PLAY → if card is magic, call `_cast_magic`; if minion, call `_deploy_minion`
- ON_SUMMON → call `resolve_summon_effect_originator`
- ON_DEATH → put card on board, kill it via direct HP=0 + `_cleanup_dead_minions`
- ON_ATTACK / ON_DAMAGED → put card on board, call `_apply_attack`
- ON_MOVE → put card on board, call `_apply_move`
- ON_DISCARD → call discard via `_apply_play_card` with a dark-matter-stash-style cost
- ON_START_OF_TURN / ON_END_OF_TURN → call `fire_start_of_turn_triggers` / `fire_end_of_turn_triggers`

3. **TestActionInvariants** — for every ActionType, fabricate a legal state in each phase and assert the contract:

```python
class TestActionInvariants:
    @pytest.mark.parametrize("phase", list(TurnPhase))
    @pytest.mark.parametrize("action_type", list(ActionType))
    def test_action_in_phase(self, library, phase, action_type):
        from grid_tactics.phase_contracts import ViolationCapture, PHASE_CONTRACTS, format_violations
        sources = _action_type_sources(action_type)  # PASS → ["action:pass_action", "action:pass_react"]; otherwise single
        # If ANY source is allowed in this phase, the action should be runnable cleanly
        allowed_in_this_phase = any(phase in PHASE_CONTRACTS[s] for s in sources)
        state = _build_state_for_action(library, phase, action_type)
        if state is None:
            pytest.skip(f"Cannot build legal state for {action_type.name} in {phase.name}")
        action = _build_action(action_type, state)
        with ViolationCapture() as cap:
            try:
                resolve_action(state, action, library)
            except Exception:
                pass  # Engine may legitimately reject the action; we're checking contract violations only
        if allowed_in_this_phase:
            assert cap.violations == [], (
                f"Action {action_type.name} in phase {phase.name}: unexpected contract violations: {format_violations(cap.violations)}"
            )
        # Note: we don't assert violations EXIST in disallowed phases, because resolve_action's dispatcher
        # may early-reject the action before it ever reaches a contract assertion. That's acceptable —
        # the dispatcher rejection IS the contract gate at that point.
```

`_build_state_for_action` is a small factory matrix mapping (phase, action_type) → minimal viable state. It returns None when the combination is impossible to construct (e.g. PLAY_REACT in START_OF_TURN with no react context). The skip count is monitored — if it skips MORE than ~30% of combos, refine the factory.

```python
    @pytest.mark.parametrize("phase", list(TurnPhase))
    @pytest.mark.parametrize("source,pending_field", list(PENDING_REQUIREMENTS.items()))
    def test_pending_action_legal_in_any_phase_when_pending_set(self, library, phase, source, pending_field):
        # Build a state in `phase` with `pending_field` set to a valid value
        state = _build_state_with_pending(library, phase, pending_field)
        if state is None:
            pytest.skip(f"Cannot construct {pending_field} in {phase.name}")
        action_type = _action_type_for_source(source)
        action = _build_action(action_type, state)
        with ViolationCapture() as cap:
            try:
                resolve_action(state, action, library)
            except OutOfPhaseError:
                pytest.fail(f"OutOfPhaseError raised for {source} in {phase.name} despite {pending_field} being set")
            except Exception:
                pass  # Engine-correctness errors are out of scope for this test
        # Pending was set → assertion should NOT fire
        assert all(v.contract_source != source for v in cap.violations), (
            f"{source} fired contract violation in {phase.name} despite {pending_field} being set: {format_violations(cap.violations)}"
        )

    @pytest.mark.parametrize("source,pending_field", list(PENDING_REQUIREMENTS.items()))
    def test_pending_action_violates_when_pending_unset(self, library, source, pending_field):
        # Same but pending_field is None — contract should fire
        state = _build_state_in_phase(library, TurnPhase.ACTION)  # any phase; pending_field is unset
        assert getattr(state, pending_field) is None
        action_type = _action_type_for_source(source)
        action = _build_action(action_type, state)
        with ViolationCapture() as cap:
            try:
                resolve_action(state, action, library)
            except Exception:
                pass
        assert any(
            v.contract_source == source and v.pending_required == pending_field
            for v in cap.violations
        ), f"Expected pending-violation for {source} when {pending_field} unset, got: {format_violations(cap.violations)}"
```
