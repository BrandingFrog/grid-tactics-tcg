```javascript
// ============================================================
// Phase 14.8: Unified Event Queue (replaces _sandboxFrameQueue,
// _pendingPostStageFrame, _pendingTriggerBlip, _pendingTurnBanner —
// all four deletions land in plan 04b once all 19 handlers are wired)
// ============================================================

var eventQueue = [];          // EngineEvent[]
var eventRunning = false;     // a slot handler is currently animating
var lastSeenSeq = -1;         // monotonic guard against re-delivery / out-of-order
                              // (server's session.next_event_seq from plan 03b M3 drives this;
                              // -1 sentinel means "no events seen yet"; reset to -1 on game_start)
var slotState = {
    spellStageChain: [],      // LIFO stack of opened-but-not-closed react windows (used by 04b)
    pendingModalKind: null,   // non-null while awaiting user input (used by 04b)
    pendingModalDeadline: 0,  // safety timeout fallback
};

function onEngineEvents(payload) {
    if (!payload || !Array.isArray(payload.events)) return;
    payload.events.forEach(function(ev) {
        if (ev.seq <= lastSeenSeq) {
            // Duplicate or out-of-order — skip
            console.warn("Skipping out-of-order event seq=" + ev.seq + " (last=" + lastSeenSeq + ")");
            return;
        }
        lastSeenSeq = ev.seq;
        eventQueue.push(ev);
    });
    // Stash final_state for snapshot-fallback rendering (e.g. on reconnect)
    if (payload.final_state) {
        window.__lastFinalState = payload.final_state;
        window.__lastLegalActions = payload.legal_actions || [];
    }
    drainEventQueue();
}

function drainEventQueue() {
    if (eventRunning) return;
    if (slotState.pendingModalKind !== null) {
        // Modal is open — wait for user input or pending_modal_resolved event (04b)
        return;
    }
    if (eventQueue.length === 0) return;
    var ev = eventQueue.shift();
    eventRunning = true;
    try {
        playEvent(ev, function onSlotDone() {
            eventRunning = false;
            commitEventToDom(ev);
            drainEventQueue();
        });
    } catch (err) {
        console.error("playEvent failed for type=" + ev.type, err);
        eventRunning = false;
        // Best-effort: commit state via final_state snapshot fallback so DOM doesn't desync
        if (window.__lastFinalState) {
            applyStateFrame(window.__lastFinalState);
        }
        drainEventQueue();
    }
}

function commitEventToDom(ev) {
    // Per-event DOM update. For events covered by full-state-snapshot rendering
    // (most engine events), we re-apply the slice that changed. For instant events
    // (mana_change, phase_changed) we patch directly. The snapshot fallback path
    // (applyStateFrame against window.__lastFinalState) is reserved for end-of-queue
    // sync after all events drain — it's NOT called per event to avoid double-render.
    // Implementation: each playEvent slot handler is RESPONSIBLE for committing its
    // own DOM mutation as part of its animation. commitEventToDom is a hook for any
    // residual mutation (e.g. updating mana counters, refreshing legal actions).
    if (ev.type === "mana_change") {
        updateManaDisplay(ev.payload.owner, ev.payload.new);
    }
    // phase_changed and game_over are 04b's responsibility (need spell-stage interaction)
    // After every event, refresh legal_actions display
    if (window.__lastLegalActions) {
        renderLegalActions(window.__lastLegalActions);
    }
}

function playEvent(ev, done) {
    // Dispatcher — maps event type to slot handler. Each handler MUST call done()
    // when its animation completes (or immediately for instant events).
    // Plan 04a implements the 10 simpler handlers; plan 04b finishes the rest.
    switch (ev.type) {
        case "minion_summoned":          return playMinionSummoned(ev, done);
        case "minion_died":              return playMinionDied(ev, done);
        case "minion_hp_change":         return playMinionHpChange(ev, done);
        case "minion_moved":             return playMinionMoved(ev, done);
        case "attack_resolved":          return playAttackResolved(ev, done);
        case "card_drawn":               return playCardDrawn(ev, done);
        case "card_played":              return playCardPlayed(ev, done);
        case "card_discarded":           return playCardDiscarded(ev, done);
        case "mana_change":              return playInstant(ev, done);
        case "player_hp_change":         return playPlayerHpChange(ev, done);
        // STUBBED until plan 04b — call done immediately so the queue drains:
        case "react_window_opened":      console.warn("react_window_opened: stub — implemented in 04b"); return setTimeout(done, 0);
        case "react_window_closed":      console.warn("react_window_closed: stub — implemented in 04b"); return setTimeout(done, 0);
        case "phase_changed":            console.warn("phase_changed: stub — implemented in 04b"); return setTimeout(done, 0);
        case "turn_flipped":             console.warn("turn_flipped: stub — implemented in 04b"); return setTimeout(done, 0);
        case "trigger_blip":             console.warn("trigger_blip: stub — implemented in 04b"); return setTimeout(done, 0);
        case "pending_modal_opened":     console.warn("pending_modal_opened: stub — implemented in 04b"); return setTimeout(done, 0);
        case "pending_modal_resolved":   console.warn("pending_modal_resolved: stub — implemented in 04b"); return setTimeout(done, 0);
        case "fizzle":                   console.warn("fizzle: stub — implemented in 04b"); return setTimeout(done, 0);
        case "game_over":                console.warn("game_over: stub — implemented in 04b"); return setTimeout(done, 0);
        default:
            console.warn("Unknown event type: " + ev.type);
            return setTimeout(done, 0);
    }
}

function playInstant(ev, done) {
    // Zero-duration events — commit happens in commitEventToDom.
    setTimeout(done, 0);
}
```
