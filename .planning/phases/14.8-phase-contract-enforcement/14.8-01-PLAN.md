5. **assert_phase_contract(state: GameState, source: str) -> None**:

   ```python
   def assert_phase_contract(state, source):
       mode = get_enforcement_mode()
       if mode == "off":
           return
       # Sandbox category bypasses phase enforcement entirely (orchestrator decision #5)
       if source.startswith("sandbox:"):
           return
       # Pending-bound actions: check pending FIRST (orchestrator decision #8)
       if source in PENDING_REQUIREMENTS:
           field = PENDING_REQUIREMENTS[source]
           if getattr(state, field, None) is None:
               err = OutOfPhaseError(...)  # pending_required=field
               if mode == "shadow":
                   _log_violation(err); return
               raise err
           return  # pending was set → action legitimate, skip phase check
       # Standard phase check
       allowed = PHASE_CONTRACTS.get(source)
       if allowed is None:
           # Unknown source — surface loudly (likely a typo / new code path missed the table)
           err = OutOfPhaseError(contract_source=source, phase=state.phase, allowed_phases=frozenset(), pending_required=None)
           err.unknown_source = True
           if mode == "shadow":
               _log_violation(err); return
           raise err
       if state.phase not in allowed:
           err = OutOfPhaseError(contract_source=source, phase=state.phase, allowed_phases=allowed, pending_required=None)
           if mode == "shadow":
               _log_violation(err); return
           raise err
   ```
