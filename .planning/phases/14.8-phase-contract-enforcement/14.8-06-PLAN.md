```python
"""EventTimeline — captures engine_events socket frames and per-event timing.

Usage in Playwright test:
    timeline = EventTimeline()
    page.on("websocket", timeline.attach_ws_listener)
    # run scenario
    timeline.assert_event_ordering(expected_events)
    timeline.assert_no_overlap(["react_window_opened", "react_window_closed"])
"""
from dataclasses import dataclass, field
from typing import Any, Optional

@dataclass
class CapturedEvent:
    seq: int
    type: str
    contract_source: str
    payload: dict
    received_at_ms: float
    started_at_ms: Optional[float] = None  # set by client-side hook
    ended_at_ms: Optional[float] = None

@dataclass
class EventTimeline:
    events: list[CapturedEvent] = field(default_factory=list)

    def attach_ws_listener(self, ws):
        """Wire to playwright page.on('websocket', ...). Capture both directions."""
        ws.on("framereceived", self._on_frame)

    def _on_frame(self, frame):
        # frame.payload is bytes/str; parse Socket.IO format → engine_events payload
        ...

    def attach_client_timing(self, page):
        """Inject a hook into game.js's playEvent done callback that records
        timing back to the page. The test reads window.eventQueueDebug.timeline
        at the end of the scenario."""
        page.evaluate("""
            window.eventQueueDebug = window.eventQueueDebug || {};
            window.eventQueueDebug.enabled = true;
            window.eventQueueDebug.timeline = [];
            const origPlayEvent = window.playEvent;
            window.playEvent = function(ev, done) {
                const start = performance.now();
                origPlayEvent(ev, function() {
                    const end = performance.now();
                    window.eventQueueDebug.timeline.push({
                        seq: ev.seq, type: ev.type,
                        started_at_ms: start, ended_at_ms: end
                    });
                    done();
                });
            };
        """)

    def merge_client_timing(self, page):
        """Pull window.eventQueueDebug.timeline back and merge into self.events."""
        client_timeline = page.evaluate("window.eventQueueDebug.timeline")
        for entry in client_timeline:
            for ev in self.events:
                if ev.seq == entry["seq"]:
                    ev.started_at_ms = entry["started_at_ms"]
                    ev.ended_at_ms = entry["ended_at_ms"]

    def assert_event_ordering(self, expected: list[dict]):
        """Assert observed events match expected (in order, by type + contract_source).
        Tolerates extra events between expected ones (e.g. mana_change). Use payload_keys
        to verify each expected event has at least the listed keys in payload."""
        idx = 0
        for ev in self.events:
            if idx < len(expected) and ev.type == expected[idx]["type"]:
                if ev.contract_source == expected[idx]["contract_source"]:
                    for key in expected[idx].get("payload_keys", []):
                        assert key in ev.payload, f"Event seq={ev.seq} type={ev.type} missing payload key '{key}'"
                    idx += 1
        assert idx == len(expected), f"Only matched {idx}/{len(expected)} expected events; missing: {expected[idx:]}"

    def assert_no_overlap(self, type_pairs: list[tuple[str, str]]):
        """For each (a, b) pair, assert no instance of b began before the latest instance of a ended."""
        for a_type, b_type in type_pairs:
            a_events = [e for e in self.events if e.type == a_type and e.ended_at_ms is not None]
            b_events = [e for e in self.events if e.type == b_type and e.started_at_ms is not None]
            # Pair them by seq (a always emits before b in correct ordering)
            for b in b_events:
                preceding_a = [a for a in a_events if a.seq < b.seq]
                if preceding_a:
                    latest_a = max(preceding_a, key=lambda x: x.ended_at_ms)
                    assert latest_a.ended_at_ms <= b.started_at_ms, (
                        f"Overlap: {a_type} (seq={latest_a.seq}, ended={latest_a.ended_at_ms}) "
                        f"did not finish before {b_type} (seq={b.seq}, started={b.started_at_ms})"
                    )
```
