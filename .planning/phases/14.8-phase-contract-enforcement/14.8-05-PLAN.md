**phase_contracts.py:** Change `get_enforcement_mode()` to default to "shadow" if env var is unset (was "off"). Production server runs at shadow by default — observability without rejection. CI explicitly sets strict.

```python
def get_enforcement_mode() -> str:
    """Returns the current enforcement mode.

    Defaults to 'shadow' in production (logs violations without raising).
    CI sets CONTRACT_ENFORCEMENT_MODE=strict via test fixtures.
    Set to 'off' explicitly to disable entirely (e.g. for legacy debugging tools).
    """
    global _cached_mode
    if _cached_mode is not None:
        return _cached_mode
    raw = os.environ.get("CONTRACT_ENFORCEMENT_MODE", "shadow").lower()
    if raw not in ("off", "shadow", "strict"):
        logging.getLogger("grid_tactics.phase_contracts").warning(
            "Unknown CONTRACT_ENFORCEMENT_MODE=%r — defaulting to shadow", raw
        )
        raw = "shadow"
    _cached_mode = raw
    return _cached_mode
```

    except OutOfPhaseError as e:
        # Strict-mode contract violation — roll back, emit structured error, continue serving
        logging.getLogger("grid_tactics.server.events").error(
            "OutOfPhaseError in handle_submit_action: source=%s phase=%s allowed=%s pending_required=%s",
            e.contract_source, e.phase, e.allowed_phases, e.pending_required,
        )
        # Rollback (M4): session state is already saved_state since resolve_action raised before
        # any commit. No explicit assignment needed if we don't reassign; but be defensive:
        # session.state = saved_state  (only if assignment is the only mutation path)
        emit('error', {

            'contract_source': e.contract_source,
            'phase': e.phase.name if e.phase else None,
            'allowed_phases': [p.name for p in (e.allowed_phases or [])],
            'pending_required': e.pending_required,
            'message': f"Action rejected: contract violation ({e.contract_source} not allowed in {e.phase.name if e.phase else 'unknown'} phase).",
        }, room=request.sid)
        return  # state unchanged; no engine_events emitted
    except Exception as e:
        # Existing catch-all unchanged — rollback to saved_state and emit generic error
        ...
    # Existing post-success path: emit engine_events, update session.next_event_seq, etc.
    session.next_event_seq = stream.next_seq
    for viewer_idx, sid in session.viewer_sids():
        filtered = filter_engine_events_for_viewer(stream.events, viewer_idx)
        emit('engine_events', {
            'events': [e.to_dict() for e in filtered],
