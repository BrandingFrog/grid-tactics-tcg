```python
"""Engine event stream — phase 14.8 wire format.

Replaces the post-resolution state-snapshot diff approach. Each state mutation
emits an EngineEvent with contract_source, payload, and animation timing. The
client's eventQueue (plan 14.8-04a/b) consumes these in seq order.

Sandbox edits (cheat_mana, undo, set_active, etc.) ALSO produce events tagged
contract_source='sandbox:<verb>'. They bypass phase-contract assertions
(see phase_contracts.py) but flow through the same event-stream pipeline so the
client uses ONE rendering path for all server-driven state changes.
"""
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Optional, Any

# Event type constants — all 19 from research §"Event types proposed"
EVT_MINION_SUMMONED = "minion_summoned"
EVT_MINION_DIED = "minion_died"
EVT_MINION_HP_CHANGE = "minion_hp_change"
EVT_MINION_MOVED = "minion_moved"
EVT_ATTACK_RESOLVED = "attack_resolved"
EVT_CARD_DRAWN = "card_drawn"
EVT_CARD_PLAYED = "card_played"
EVT_CARD_DISCARDED = "card_discarded"
EVT_MANA_CHANGE = "mana_change"
EVT_PLAYER_HP_CHANGE = "player_hp_change"
EVT_REACT_WINDOW_OPENED = "react_window_opened"
EVT_REACT_WINDOW_CLOSED = "react_window_closed"
EVT_PHASE_CHANGED = "phase_changed"
EVT_TURN_FLIPPED = "turn_flipped"
EVT_TRIGGER_BLIP = "trigger_blip"
EVT_PENDING_MODAL_OPENED = "pending_modal_opened"
EVT_PENDING_MODAL_RESOLVED = "pending_modal_resolved"
EVT_FIZZLE = "fizzle"
EVT_GAME_OVER = "game_over"

ALL_EVENT_TYPES = frozenset({
    EVT_MINION_SUMMONED, EVT_MINION_DIED, EVT_MINION_HP_CHANGE, EVT_MINION_MOVED,
    EVT_ATTACK_RESOLVED, EVT_CARD_DRAWN, EVT_CARD_PLAYED, EVT_CARD_DISCARDED,
    EVT_MANA_CHANGE, EVT_PLAYER_HP_CHANGE, EVT_REACT_WINDOW_OPENED,
    EVT_REACT_WINDOW_CLOSED, EVT_PHASE_CHANGED, EVT_TURN_FLIPPED, EVT_TRIGGER_BLIP,
    EVT_PENDING_MODAL_OPENED, EVT_PENDING_MODAL_RESOLVED, EVT_FIZZLE, EVT_GAME_OVER,
})

# Default animation durations (ms) — client uses event-provided value if set, else this lookup
DEFAULT_DURATION_MS = {
    EVT_MINION_SUMMONED: 600,
    EVT_MINION_DIED: 0,                    # covered by trigger blip / hp_popup
    EVT_MINION_HP_CHANGE: 400,             # floating popup
    EVT_MINION_MOVED: 350,
    EVT_ATTACK_RESOLVED: 500,
    EVT_CARD_DRAWN: 350,
    EVT_CARD_PLAYED: 0,                    # covered by spell stage
    EVT_CARD_DISCARDED: 300,
    EVT_MANA_CHANGE: 0,
    EVT_PLAYER_HP_CHANGE: 400,
    EVT_REACT_WINDOW_OPENED: 600,          # spell stage in
    EVT_REACT_WINDOW_CLOSED: 400,          # spell stage out
    EVT_PHASE_CHANGED: 0,
    EVT_TURN_FLIPPED: 1500,                # turn banner duration
    EVT_TRIGGER_BLIP: 900,                 # source pulse + center icon + target pulse
    EVT_PENDING_MODAL_OPENED: 0,           # gates queue, no fixed duration (waits for input)
    EVT_PENDING_MODAL_RESOLVED: 0,
    EVT_FIZZLE: 350,                       # optional puff
    EVT_GAME_OVER: 0,                      # game-over modal handles its own timing
}

@dataclass
class EngineEvent:
    type: str
    contract_source: str
    seq: int
    payload: dict
    animation_duration_ms: int = 0
    triggered_by_seq: Optional[int] = None
    requires_decision: bool = False

    def to_dict(self) -> dict:
        d = {
            "type": self.type,
            "contract_source": self.contract_source,
            "seq": self.seq,
            "payload": self.payload,
            "animation_duration_ms": self.animation_duration_ms,
            "triggered_by_seq": self.triggered_by_seq,
            "requires_decision": self.requires_decision,
        }
        return d

    @classmethod
    def from_dict(cls, d: dict) -> "EngineEvent":
        return cls(
            type=d["type"],
            contract_source=d["contract_source"],
            seq=d["seq"],
            payload=d["payload"],
            animation_duration_ms=d.get("animation_duration_ms", 0),
            triggered_by_seq=d.get("triggered_by_seq"),
            requires_decision=d.get("requires_decision", False),
        )


@dataclass
class EventStream:
    """Collector handed into resolve_action. Append-only event log per call.

    Per-game seq counter is owned by the SESSION (server-side); each EventStream
    instance gets a starting seq from the session's next_event_seq field
    (added in plan 14.8-03b) and increments locally. After the call, the session
    persists stream.next_seq back to its next_event_seq field.

    Plan 14.8-08 (deferred to Phase 15) uses seq for reconnect/replay.
    """
    next_seq: int = 0
    events: list = field(default_factory=list)
    _parent_seq_stack: list = field(default_factory=list)  # for triggered_by_seq nesting

    def collect(
        self,
        type: str,
        contract_source: str,
        payload: dict,
        *,
        animation_duration_ms: Optional[int] = None,
        requires_decision: bool = False,
    ) -> EngineEvent:
        assert type in ALL_EVENT_TYPES, f"Unknown event type: {type}"
        if animation_duration_ms is None:
            animation_duration_ms = DEFAULT_DURATION_MS[type]
        triggered_by = self._parent_seq_stack[-1] if self._parent_seq_stack else None
        ev = EngineEvent(
            type=type,
            contract_source=contract_source,
            seq=self.next_seq,
            payload=payload,
            animation_duration_ms=animation_duration_ms,
            triggered_by_seq=triggered_by,
            requires_decision=requires_decision,
        )
        self.events.append(ev)
        self.next_seq += 1
        return ev

    def push_parent(self, seq: int) -> None:
        """Inline trigger nesting (per research pitfall §4): wrap a resolve_effect
        call so events emitted inside have triggered_by_seq pointing to the parent."""
        self._parent_seq_stack.append(seq)

    def pop_parent(self) -> None:
        self._parent_seq_stack.pop()

    def to_dict_list(self) -> list[dict]:
        return [e.to_dict() for e in self.events]
```

**Thread the collector through resolve_action call sites:**

The threading rule is: any function that already received `assert_phase_contract` in plan 01 ALSO accepts `event_collector: EventStream | None = None` as a kwarg. Default None means no events emitted (back-compat).

- `resolve_action(state, action, library, *, event_collector=None)` — top-level entry. Emits action-level events directly, threads collector to inner calls.
- `resolve_effect(..., event_collector=None)`, `resolve_effects_for_trigger(..., event_collector=None)`, `_resolve_conjure(..., event_collector=None)`, `_apply_effect_to_minion(..., event_collector=None)`, `_apply_effect_to_player(..., event_collector=None)` — all gain the kwarg.

For each engine site, emit the matching event RIGHT AFTER the `replace(state, ...)` returns the new state. Use the new state's values in the payload. Examples:

- `_apply_damage_to_minion`: emit `EVT_MINION_HP_CHANGE` with payload `{instance_id, prev_hp, new_hp, delta, owner_idx, position}`.
- `_apply_attack`: emit `EVT_ATTACK_RESOLVED` with payload `{attacker_id, defender_id, attacker_dmg, defender_dmg, attacker_killed, defender_killed, positions}`. The inline ON_ATTACK / ON_DAMAGED nested calls use `push_parent(attack_event.seq)` / `pop_parent()` to scope their events as children.
- `_cleanup_dead_minions`: emit one `EVT_MINION_DIED` per killed minion with `{instance_id, position, owner_idx, card_numeric_id}`.
- `_apply_move`: emit `EVT_MINION_MOVED` with `{instance_id, from, to, owner_idx}`.
- `_deploy_minion`: emit `EVT_MINION_SUMMONED` AFTER the originator resolves at the bottom of the react chain (NOT at deploy time — the minion isn't on the board until summon-effect-resolves, per Phase 14.7 cast-mode-originator semantics).
- `_apply_play_card`: emit `EVT_CARD_PLAYED` with `{card_id, owner, target_pos}` at the start; emit `EVT_CARD_DISCARDED` for each discard cost.
- `_apply_pass` (fatigue): emit `EVT_PLAYER_HP_CHANGE` for fatigue damage.
- `_check_game_over`: when game_over is set, emit `EVT_GAME_OVER` with `{winner, reason}`.
- `tick_status_effects`: per minion that takes burn damage, emit `EVT_MINION_HP_CHANGE` with `contract_source="status:burn"`.
