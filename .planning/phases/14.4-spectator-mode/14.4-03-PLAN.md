  @socketio.on("spectate_room")
  def handle_spectate_room(data):
      display_name = (data or {}).get("display_name", "").strip()
      room_code = (data or {}).get("room_code", "").strip().upper()
      god_mode = bool((data or {}).get("god_mode", False))
      if not display_name:
          emit("error", {"msg": "display_name is required"}); return
      if not room_code:
          emit("error", {"msg": "room_code is required"}); return
      try:
          token, room = room_manager.join_as_spectator(room_code, display_name, request.sid, god_mode)
      except ValueError as e:
          emit("error", {"msg": str(e)}); return
      join_room(room_code)  # flask-socketio room join for any room-wide broadcasts
      emit("spectator_joined", {
          "session_token": token,
          "room_code": room_code,
          "god_mode": god_mode,
      })
      # If a game is already in progress, immediately send current state
      session = room_manager.get_game(room_code)
      if session is not None:
          state = session.current_state_dict()  # use whatever existing accessor returns the dict form
          spec_state = filter_state_for_spectator(state, god_mode=god_mode, perspective_idx=0)
          emit("state_update", {"state": spec_state, "resolved_action": None})
