    spectatorGodMode = data.god_mode;
    sessionToken = data.session_token;
    roomCode = data.room_code;
    // Transition to game screen using whatever existing function does this for players
    showGameScreen();  // or the actual function name — find it (likely showScreen('game') or similar)
  });
