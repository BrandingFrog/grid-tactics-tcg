```python
class SandboxSession:
    """Single-tab manual game state editor. Wraps the existing engine; never duplicates it.

    State invariant: every public mutator pushes the OLD state onto _undo before
    advancing _state. _redo is cleared on any non-undo/redo mutation. Undo/redo
    operate on whole-state snapshots (immutable GameState references) so no
    surgery is needed for mid-react / mid-tutor / mid-pending positions.
    """

    def __init__(self, library: CardLibrary, sid: str):
        self.library = library
        self.sid = sid
        # Fresh empty-deck game so the user builds the scenario from scratch.
        # Empty decks are valid input to GameState.new_game (deck shuffle is a no-op
        # over empty tuples). The starting-hand-draw loop draws zero cards.
        self._state, self._rng = GameState.new_game(seed=0, deck_p1=(), deck_p2=())
        self._active_view_idx: int = 0  # which player the client is currently "controlling"
        self._undo: deque[GameState] = deque(maxlen=HISTORY_MAX)
        self._redo: deque[GameState] = deque(maxlen=HISTORY_MAX)
        self.lock = threading.Lock()

    @property
    def state(self) -> GameState:
        return self._state

    @property
    def active_view_idx(self) -> int:
        return self._active_view_idx

    # ---- mutation primitives ----------------------------------------------

    def _push_undo(self) -> None:
        self._undo.append(self._state)
        self._redo.clear()

    def apply_action(self, action: Action) -> None:
        """Validate via legal_actions(), apply via resolve_action(). Raises ValueError if illegal."""
        valid = legal_actions(self._state, self.library)
        if action not in valid:
            raise ValueError("Illegal action")
        self._push_undo()
        self._state = resolve_action(self._state, action, self.library)

    def add_card_to_hand(self, player_idx: int, card_numeric_id: int) -> None:
        """Sandbox-only: insert a card into a player's hand via dataclasses.replace.

        This is the ONLY new state-construction helper allowed in this phase.
        It does not bypass the engine — it just builds a new immutable GameState
        with one extra entry in Player.hand. No validation other than bounds.
        """
        if player_idx not in (0, 1):
            raise ValueError("player_idx must be 0 or 1")
        if not (0 <= card_numeric_id < self.library.card_count):
            raise ValueError("unknown card_numeric_id")
        self._push_undo()
        old_player = self._state.players[player_idx]
        new_hand = old_player.hand + (card_numeric_id,)
        new_player = replace(old_player, hand=new_hand)
        new_players = (
            (new_player, self._state.players[1]) if player_idx == 0
            else (self._state.players[0], new_player)
        )
        self._state = replace(self._state, players=new_players)

    def set_active_player(self, player_idx: int) -> None:
        """Switch which player the client is controlling. This DOES advance state
        (changes active_player_idx on the GameState) so legal_actions() reports
        the new player's options. Pushed to undo stack.
        """
        if player_idx not in (0, 1):
            raise ValueError("player_idx must be 0 or 1")
        if self._state.active_player_idx == player_idx and self._active_view_idx == player_idx:
            return
        self._push_undo()
        self._state = replace(self._state, active_player_idx=player_idx)
        self._active_view_idx = player_idx

    def undo(self) -> bool:
        if not self._undo:
            return False
        self._redo.append(self._state)
        self._state = self._undo.pop()
        return True

    def redo(self) -> bool:
        if not self._redo:
            return False
        self._undo.append(self._state)
        self._state = self._redo.pop()
        return True

    def reset(self) -> None:
        self._push_undo()
        self._state, self._rng = GameState.new_game(seed=0, deck_p1=(), deck_p2=())
        self._active_view_idx = 0

    # ---- serialization (delegates entirely to GameState) -------------------

    def to_dict(self) -> dict:
        """Serialize via the existing GameState.to_dict() — no extra fields."""
        return {
            "state": self._state.to_dict(),
            "active_view_idx": self._active_view_idx,
        }

    def load_dict(self, payload: dict) -> None:
        """Restore via the existing GameState.from_dict(). Clears undo/redo."""
        new_state = GameState.from_dict(payload["state"])
        self._state = new_state
        self._active_view_idx = int(payload.get("active_view_idx", 0))
        self._undo.clear()
        self._redo.clear()

    def legal_actions(self) -> tuple:
        """Pass-through to the engine for the current state."""
        return legal_actions(self._state, self.library)
```

- Undo/redo stores full `GameState` references. They are frozen dataclasses, so this is cheap and correct — no deep copy needed.
- The lock is held by callers (Socket.IO handlers), but include it for defensive symmetry with `GameSession.lock`.
- DO NOT inherit from `GameSession`. DO NOT import `GameSession`.

    # ------------------------------------------------------------------
    # Sandbox API (Phase 14.6)
    # ------------------------------------------------------------------
    # Sandboxes are keyed by socket SID, not by session token, and live in a
    # parallel dict so the existing room/game/spectator code paths are 100%
    # untouched. One sandbox per browser tab. No multi-user sharing.

    def create_sandbox(self, sid: str) -> SandboxSession:
        """Create (or replace) the sandbox session for this SID."""
        with self._lock:
            session = SandboxSession(self._library, sid)
            self._sandboxes[sid] = session
            return session

    def get_sandbox(self, sid: str) -> SandboxSession | None:
        """Look up the sandbox session for this SID, if any."""
        return self._sandboxes.get(sid)

    def remove_sandbox(self, sid: str) -> None:
        """Drop the sandbox for this SID (called on disconnect)."""
        with self._lock:
            self._sandboxes.pop(sid, None)
