**A. Search + add-card.** Add this to the sandbox section:

```js
function setupSandboxToolbar() {
    // Card search
    const searchInput = document.getElementById('sandbox-search');
    const resultsBox = document.getElementById('sandbox-search-results');
    if (searchInput && resultsBox) {
        searchInput.addEventListener('input', function() {
            const q = searchInput.value.trim().toLowerCase();
            if (!q) {
                resultsBox.hidden = true;
                resultsBox.innerHTML = '';
                return;
            }
            if (!sandboxCardDefs) return;
            const matches = [];
            for (const [nidStr, def] of Object.entries(sandboxCardDefs)) {
                if (def && def.name && def.name.toLowerCase().includes(q)) {
                    matches.push({ nid: parseInt(nidStr, 10), def });
                    if (matches.length >= 30) break;
                }
            }
            resultsBox.innerHTML = matches.map(function(m) {
                const cost = (m.def.mana_cost != null) ? m.def.mana_cost : '-';
                return '<div class="sandbox-search-result" data-nid="' + m.nid + '">' +
                       '<span class="sandbox-search-result-name">' + escapeHtml(m.def.name) + '</span>' +
                       '<span class="sandbox-search-result-meta">cost ' + cost + '</span>' +
                       '</div>';
            }).join('');
            resultsBox.hidden = matches.length === 0;
        });
        resultsBox.addEventListener('click', function(e) {
            const row = e.target.closest('.sandbox-search-result');
            if (!row) return;
            const nid = parseInt(row.dataset.nid, 10);
            socket.emit('sandbox_add_card', {
                player_idx: sandboxAddTargetIdx,
                card_numeric_id: nid,
            });
            // Don't clear the search — user often wants to add the same/similar card multiple times
        });
        // Hide dropdown on outside click
        document.addEventListener('click', function(e) {
            if (!e.target.closest('.sandbox-search-wrap')) {
                resultsBox.hidden = true;
            }
        });
    }

    // Target toggle (which hand to add to)
    document.querySelectorAll('.sandbox-target-btn').forEach(function(btn) {
        btn.addEventListener('click', function() {
            sandboxAddTargetIdx = parseInt(btn.dataset.target, 10);
            document.querySelectorAll('.sandbox-target-btn').forEach(function(b) {
                b.classList.toggle('active', b === btn);
            });
        });
    });

    // Control toggle (which player you ARE)
    document.querySelectorAll('.sandbox-control-btn').forEach(function(btn) {
        btn.addEventListener('click', function() {
            const idx = parseInt(btn.dataset.control, 10);
            socket.emit('sandbox_set_active_player', { player_idx: idx });
            // The active button highlight will update on the next sandbox_state via renderSandboxToolbarState
        });
    });

    // Undo / Redo
    document.getElementById('sandbox-undo-btn')?.addEventListener('click', function() {
        socket.emit('sandbox_undo');
    });
    document.getElementById('sandbox-redo-btn')?.addEventListener('click', function() {
        socket.emit('sandbox_redo');
    });

    // Reset
    document.getElementById('sandbox-reset-btn')?.addEventListener('click', function() {
        if (confirm('Reset sandbox to empty? This will clear undo history.')) {
            socket.emit('sandbox_reset');
        }
    });

    // Save (download JSON)
    document.getElementById('sandbox-save-btn')?.addEventListener('click', function() {
        socket.emit('sandbox_save');
    });

    // Load (file)
    const loadBtn = document.getElementById('sandbox-load-btn');
    const fileInput = document.getElementById('sandbox-load-file');
    if (loadBtn && fileInput) {
        loadBtn.addEventListener('click', function() { fileInput.click(); });
        fileInput.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(ev) {
                try {
                    const payload = JSON.parse(ev.target.result);
                    socket.emit('sandbox_load', { payload: payload });
                } catch (err) {
                    alert('Invalid JSON file: ' + err.message);
                }
                fileInput.value = '';
            };
            reader.readAsText(file);
        });
    }

    // Copy share code (base64 of current state)
    document.getElementById('sandbox-share-btn')?.addEventListener('click', function() {
        if (!sandboxState) return;
        const payload = { state: sandboxState, active_view_idx: sandboxActiveViewIdx };
        const code = btoa(unescape(encodeURIComponent(JSON.stringify(payload))));
        navigator.clipboard.writeText(code).then(function() {
            alert('Sandbox code copied to clipboard (' + code.length + ' chars)');
        }, function(err) {
            // Fallback for clipboard failure: prompt
            window.prompt('Copy this sandbox code:', code);
        });
    });

    // Paste share code
    document.getElementById('sandbox-paste-btn')?.addEventListener('click', function() {
        const code = window.prompt('Paste sandbox code:');
        if (!code) return;
        try {
            const payload = JSON.parse(decodeURIComponent(escape(atob(code.trim()))));
            socket.emit('sandbox_load', { payload: payload });
        } catch (err) {
            alert('Invalid sandbox code: ' + err.message);
        }
    });
}

function escapeHtml(s) {
    return String(s).replace(/[&<>"']/g, function(c) {
        return { '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;' }[c];
    });
}
```

```js
        const json = JSON.stringify(data.payload, null, 2);

        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');

    });
```

**D. Restore from localStorage on init.** Modify `initSandboxScreen` to attempt restore BEFORE creating fresh:

```js
function initSandboxScreen() {
    if (!socket || !socket.connected) {
        console.warn('[sandbox] socket not connected yet');
        return;
    }
    setupSandboxToolbar();
    // Try restore from localStorage; fall back to a fresh sandbox
    let restored = false;
    try {
        const raw = localStorage.getItem(SANDBOX_AUTOSAVE_KEY);
        if (raw) {
            const payload = JSON.parse(raw);
            // sandbox_load auto-creates if no session exists
            socket.emit('sandbox_load', { payload: payload });
            restored = true;
        }
    } catch (e) { /* corrupt — just create fresh */ }
    if (!restored) {
        socket.emit('sandbox_create');
    }
}
```

**E. Action submission — REUSE the live game's click handlers.**

This is the highest-risk integration point. The live game has click handlers on board cells and on hand cards that:
1. Track a selection state (selected card / selected minion)
2. Highlight valid targets from the legal_actions list
3. On valid-target click, build an Action and call `submitAction(action)` which emits `submit_action`

For the sandbox, the SAME flow must run, but the final emit must go to `sandbox_apply_action` instead of `submit_action`. The cleanest implementation is to add a single conditional in the existing `submitAction` (or equivalent) function:

Find the existing emit site (grep for `socket.emit('submit_action'` in game.js) and wrap it:

```js
function submitAction(action) {
    if (sandboxMode) {
        socket.emit('sandbox_apply_action', serializeAction(action));
    } else {
        socket.emit('submit_action', serializeAction(action));
    }
}
```

If the existing function name differs (e.g. `emitAction`, `sendAction`, `submit`), use that name. The change is ONE conditional. Do NOT duplicate any targeting / highlighting / serialization logic.

For the legal-actions source, the existing click handlers read from a module variable (likely `legalActions` or `currentLegalActions`). When `sandboxMode === true`, those handlers must read from `sandboxLegalActions` instead. Add a single accessor at the top of the targeting code (or a shim function):

```js
function getActiveLegalActions() {
    return sandboxMode ? sandboxLegalActions : (legalActions || currentLegalActions || []);
}
```

Then in the existing click handlers, replace direct reads of `legalActions` / `currentLegalActions` with `getActiveLegalActions()`. ONLY in the click-targeting code path. Do NOT touch other reads (e.g. live-game state listeners).

Similarly for the `state` and `your_player_idx` references inside the click handlers, expose:
```js
function getActiveGameState() {
    return sandboxMode ? sandboxState : gameState;
}
function getActivePlayerIdx() {
    return sandboxMode ? sandboxState.active_player_idx : yourPlayerIdx;
}
```

And replace inside the click handlers. The exact set of replacements depends on the existing code — the executor must read game.js carefully and apply the minimum number of substitutions to make the click handlers state-source-agnostic.

**Important rule:** if the existing click handlers cannot be cleanly retargeted with <20 lines of substitutions, the executor MUST stop and emit a checkpoint asking the user how to proceed — do NOT duplicate the click flow into a parallel sandbox-only set of handlers.
