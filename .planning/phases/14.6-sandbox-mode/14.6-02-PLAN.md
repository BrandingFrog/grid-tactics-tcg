**Change 3: Sandbox initializer + state listener + render path.** Add a new section near the bottom of game.js (after the existing screen handlers, before any final IIFE or `window.addEventListener('load', ...)` block). This is the entire sandbox client surface for plan 14.6-02:

```js
// =============================================
// Section: Sandbox Mode (Phase 14.6)
// =============================================

function initSandboxScreen() {
    // Always create a fresh sandbox on (re)entering the screen — one per tab.
    if (!socket || !socket.connected) {
        console.warn('[sandbox] socket not connected yet');
        return;
    }
    socket.emit('sandbox_create');
}

function setupSandboxSocketHandlers() {
    if (!socket) return;

    socket.on('sandbox_card_defs', function(data) {
        sandboxCardDefs = data.card_defs || {};
        // Mirror into allCardDefs only if it isn't already populated by the live game,
        // so the existing renderers (which read allCardDefs) work inside the sandbox.
        if (!allCardDefs) {
            allCardDefs = sandboxCardDefs;
        }
    });

    socket.on('sandbox_state', function(data) {
        sandboxState = data.state;
        sandboxLegalActions = data.legal_actions || [];
        sandboxActiveViewIdx = data.active_view_idx || 0;
        sandboxUndoDepth = data.undo_depth || 0;
        sandboxRedoDepth = data.redo_depth || 0;
        renderSandbox();
    });

    socket.on('sandbox_save_blob', function(data) {
        // Plan 14.6-03 will wire this to download / share-code / localStorage.
        // For plan 14.6-02 we just stash it on window for manual smoke testing.
        window.__lastSandboxBlob = data.payload;
    });
}

function renderSandbox() {
    if (!sandboxState || !sandboxCardDefs) return;

    // Renderer reuse contract: build a synthetic frame matching the same shape
    // the live game renders from, then call the SAME render functions that
    // state_update uses. The sandbox screen's DOM ids are sandbox-prefixed, so
    // we temporarily point the renderers at them via a mount-target switch.
    //
    // Strategy: most existing renderers take a parent element argument or read
    // from well-known IDs. To avoid duplicating logic, we re-use renderBoard /
    // renderHand / renderStats by passing the sandbox mount points as the target.
    // If a renderer hard-codes IDs, this task wraps it in a mount-swap helper.

    renderSandboxBoard();
    renderSandboxHands();
    renderSandboxStats();
}

function renderSandboxBoard() {
    // Reuse the live game's board renderer. Find the existing function used by
    // applyStateFrame / state_update — it is `renderBoard(state)` (or whatever
    // the canonical name is in this codebase). Locate it via grep before this
    // task and call it here with the sandbox mount target.
    //
    // Implementation note for executor: if the existing renderBoard reads from
    // `gameState` and writes to a hard-coded `#board` element, the minimal
    // additive change is to:
    //   (a) temporarily set window.gameState = sandboxState
    //   (b) temporarily move/swap the #board element OR pass an explicit target
    //   (c) restore window.gameState after render
    //
    // The cleanest version is to refactor renderBoard to accept (state, target)
    // arguments — IF and ONLY IF the existing call sites can pass undefined to
    // get the legacy behavior. This is the ONE small refactor allowed.
    const target = document.getElementById('sandbox-board');
    if (!target) return;
    if (typeof window.renderBoard === 'function') {
        // Preferred: explicit mount target
        try {
            window.renderBoard(sandboxState, { mount: target, perspectiveIdx: sandboxActiveViewIdx });
            return;
        } catch (e) {
            // Fall through to legacy mount swap
        }
    }
    // Legacy fallback: clone the live game's board element into our mount.
    // This is a structural mount, not a re-render — plan 14.6-03 may revisit.
    const legacyBoard = document.getElementById('board');
    if (legacyBoard) {
        const prevState = window.gameState;
        window.gameState = sandboxState;
        // Trigger whatever render function exists in this codebase. Common names:
        // renderBoard(), updateBoard(), drawBoard(). Executor: pick the right one
        // by reading game.js. The call must mutate the existing live #board, then
        // we move its children into #sandbox-board.
        if (typeof window.renderBoard === 'function') window.renderBoard();
        else if (typeof window.updateBoard === 'function') window.updateBoard();
        else if (typeof window.drawBoard === 'function') window.drawBoard();
        target.innerHTML = legacyBoard.innerHTML;
        window.gameState = prevState;
    }
}

function renderSandboxHands() {
    // Same reuse contract as renderSandboxBoard. The sandbox is god view, so we
    // render BOTH players' hands. Mount targets: #sandbox-hand-p0 and #sandbox-hand-p1.
    const p0Target = document.getElementById('sandbox-hand-p0');
    const p1Target = document.getElementById('sandbox-hand-p1');
    if (!p0Target || !p1Target) return;

    if (typeof window.renderHand === 'function') {
        try {
            window.renderHand(sandboxState.players[0].hand, { mount: p0Target, ownerIdx: 0, faceUp: true });
            window.renderHand(sandboxState.players[1].hand, { mount: p1Target, ownerIdx: 1, faceUp: true });
            return;
        } catch (e) {
            // Fall back to manual rendering using renderCardFrame (the unified card renderer
            // shipped in Phase 14.5 — it MUST exist, find it via grep).
        }
    }
    // Fallback path: use renderCardFrame directly.
    if (typeof window.renderCardFrame === 'function') {
        p0Target.innerHTML = '';
        p1Target.innerHTML = '';
        for (const cardId of sandboxState.players[0].hand) {
            const def = sandboxCardDefs[cardId];
            if (def) p0Target.appendChild(window.renderCardFrame(def));
        }
        for (const cardId of sandboxState.players[1].hand) {
            const def = sandboxCardDefs[cardId];
            if (def) p1Target.appendChild(window.renderCardFrame(def));
        }
    }
}

function renderSandboxStats() {
    const target = document.getElementById('sandbox-stats');
    if (!target) return;
    // Lightweight HP/mana display. The full live-game stats renderer can be
    // wired in plan 14.6-03 if needed. For plan 14.6-02 a minimal display
    // proves the data path; styling matches live-game classes for visual parity.
    const p0 = sandboxState.players[0];
    const p1 = sandboxState.players[1];
    target.innerHTML = `
        <div class="player-stats">
            <strong>P1</strong>
            <span>HP ${p0.hp}</span>
            <span>Mana ${p0.current_mana}/${p0.max_mana}</span>
            <span>Hand ${p0.hand.length}</span>
            <span>Deck ${p0.deck.length}</span>
        </div>
        <div class="player-stats">
            <strong>Active: P${(sandboxState.active_player_idx ?? 0) + 1}</strong>
            <span>Phase ${sandboxState.phase}</span>
            <span>Turn ${sandboxState.turn_number}</span>
        </div>
        <div class="player-stats">
            <strong>P2</strong>
            <span>HP ${p1.hp}</span>
            <span>Mana ${p1.current_mana}/${p1.max_mana}</span>
            <span>Hand ${p1.hand.length}</span>
            <span>Deck ${p1.deck.length}</span>
        </div>
    `;
}
```
